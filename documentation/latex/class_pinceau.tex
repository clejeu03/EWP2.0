    painter.setBrush(brush);
    painter.setPen(Qt::NoPen);

    static const int points[] = {5,0,(5/2),(5/2),0,5,0,15,300,15,300,5,(295+(5/
      2)),(5/2),295,0};
    QPolygon polygon;

    painter.drawPolygon(polygon, Qt::OddEvenFill);

    //dessin des actions
